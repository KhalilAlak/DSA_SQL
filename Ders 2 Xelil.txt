       hire_date,
       EXTRACT(YEAR FROM hire_date) date_year1,

       to_char(hire_date,'MM') date_month2,

SELECT first_name||' '||last_name,
       salary,
       CASE WHEN salary>15000 THEN 'Varli'
            WHEN salary BETWEEN 10000 AND 15000 THEN 'Orta'
            WHEN salary BETWEEN 5000 AND 10000 THEN 'Kasib'
            WHEN salary<5000 THEN 'Yoxsul' 
            ELSE 'N/A' END AS metric
FROM hr.employees;
